import requests
from bs4 import BeautifulSoup
from IPython.display import Image, display
import random

website = "https://scikit-learn.org/stable/tutorial/statistical_inference/supervised_learning.html"
images_url = "https://scikit-learn.org/stable/"

response = requests.get(website)

html_data = BeautifulSoup(response.text, "html.parser")
images_list = html_data.find_all("img")

image_names = []

for image_tag in images_list:
    image_folder = image_tag["src"].split("/")[-2]
    image_name = image_tag["src"].split("/")[-1]

    image_names.append(image_name)

    download_url = images_url + image_folder + "/" + image_name

    image_data = requests.get(download_url).content

    image_file = open(image_name, "wb")
    image_file.write(image_data)
    image_file.close()

print("Download ended")

display(Image(image_names[random.randint(0, len(image_names) - 1)]))
